local function GetClosestPlayer()
    local Target = nil
    local MaxDistance = State.AimbotFOV
    for _, Player in ipairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character and Player.Character:FindFirstChild(State.AimbotPart) and Player.Character:FindFirstChildOfClass("Humanoid") and Player.Character:FindFirstChildOfClass("Humanoid").Health > 0 then
            local ScreenPoint, OnScreen = Camera:WorldToScreenPoint(Player.Character[State.AimbotPart].Position)
            if OnScreen then
                local MousePos = UserInputService:GetMouseLocation()
                local Distance = (Vector2.new(ScreenPoint.X, ScreenPoint.Y) - MousePos).Magnitude
                if Distance < MaxDistance then
                    Target = Player
                    MaxDistance = Distance
                end
            end
        end
    end
    return Target
end

local function ApplyHighlight(Player)
    if Player.Character then
        local Highlight = Player.Character:FindFirstChildOfClass("Highlight")
        if not Highlight then
            Highlight = Instance.new("Highlight")
            Highlight.Parent = Player.Character
        end
        Highlight.Adornee = Player.Character
        Highlight.FillColor = State.ESP_Color
        Highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
        Highlight.FillTransparency = State.ESP_Box and 0.5 or 1
        Highlight.OutlineTransparency = State.ESP_Box and 0 or 1
        Highlight.Enabled = true
    end
end

local function RemoveHighlight(Player)
    if Player.Character then
        local Highlight = Player.Character:FindFirstChildOfClass("Highlight")
        if Highlight then
            Highlight:Destroy()
        end
    end
end

local FreecamPart = nil
local FreecamActive = false

local function ToggleFreecam(Val)
    FreecamActive = Val
    if FreecamActive then
        FreecamPart = Instance.new("Part")
        FreecamPart.Size = Vector3.new(1, 1, 1)
        FreecamPart.Transparency = 1
        FreecamPart.CanCollide = false
        FreecamPart.Anchored = true
        FreecamPart.CFrame = Camera.CFrame
        FreecamPart.Parent = Workspace
        Camera.CameraSubject = FreecamPart
    else
        if FreecamPart then
            FreecamPart:Destroy()
            FreecamPart = nil
        end
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid") then
            Camera.CameraSubject = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        end
    end
end

local TracersFolder = Instance.new("Folder", Workspace)
TracersFolder.Name = "AppleCat_Tracers"

RunService.RenderStepped:Connect(function()
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid") then
        local Hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        Hum.WalkSpeed = State.WalkSpeed
        Hum.JumpPower = State.JumpPower
    end
    
    if State.Fly and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local HRP = LocalPlayer.Character.HumanoidRootPart
        local MoveDir = LocalPlayer.Character:FindFirstChildOfClass("Humanoid").MoveDirection
        HRP.Velocity = MoveDir * State.FlySpeed + Vector3.new(0, 0.5, 0)
    end
    
    if State.SpinBot and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local HRP = LocalPlayer.Character.HumanoidRootPart
        HRP.CFrame = HRP.CFrame * CFrame.Angles(0, math.rad(State.SpinSpeed / 10), 0)
    end
    
    if State.Jesus and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local HRP = LocalPlayer.Character.HumanoidRootPart
        local RayCast = RaycastParams.new()
        RayCast.FilterType = Enum.RaycastFilterType.Blacklist
        RayCast.FilterDescendantsInstances = {LocalPlayer.Character}
        local Result = Workspace:Raycast(HRP.Position, Vector3.new(0, -3.5, 0), RayCast)
        if Result and (Result.Material == Enum.Material.Water) then
            HRP.Velocity = Vector3.new(HRP.Velocity.X, 0, HRP.Velocity.Z)
            HRP.CFrame = HRP.CFrame + Vector3.new(0, -Result.Position.Y + HRP.Position.Y - 3, 0)
        end
    end
    
    if State.InfOxygen and LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid") then
        LocalPlayer.Character:FindFirstChildOfClass("Humanoid").MaxOxygen = 99999
        LocalPlayer.Character:FindFirstChildOfClass("Humanoid").Oxygen = 99999
    end
    
    if State.Aimbot then
        local Target = GetClosestPlayer()
        if Target and Target.Character and Target.Character:FindFirstChild(State.AimbotPart) then
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, Target.Character[State.AimbotPart].Position)
        end
    end
    
    if State.DrawFOV then
        State.FOVCircle.Position = UserInputService:GetMouseLocation()
    end
    
    if State.HitboxExpander then
        for _, P in ipairs(Players:GetPlayers()) do
            if P ~= LocalPlayer and P.Character and P.Character:FindFirstChild("HumanoidRootPart") then
                local Part = P.Character.HumanoidRootPart
                Part.Size = Vector3.new(State.HitboxSize, State.HitboxSize, State.HitboxSize)
                Part.Transparency = 0.7
                Part.Color = Color3.fromRGB(255, 0, 0)
                Part.CanCollide = false
            end
        end
    end
    
    if State.KillAura then
        for _, P in ipairs(Players:GetPlayers()) do
            if P ~= LocalPlayer and P.Character and P.Character:FindFirstChild("HumanoidRootPart") and P.Character:FindFirstChildOfClass("Humanoid") and P.Character:FindFirstChildOfClass("Humanoid").Health > 0 then
                local HRP = P.Character.HumanoidRootPart
                local LocalHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if LocalHRP and (HRP.Position - LocalHRP.Position).Magnitude <= State.KillAuraRange then
                    local Tool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
                    if Tool then
                        Tool:Activate()
                    end
                end
            end
        end
    end
    
    if State.TriggerBot then
        local Mouse = LocalPlayer:GetMouse()
        if Mouse.Target and Mouse.Target.Parent then
            local Player = Players:GetPlayerFromCharacter(Mouse.Target.Parent)
            if Player and Player ~= LocalPlayer then
                local Tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
                if Tool then
                    Tool:Activate()
                end
            end
        end
    end
    
    TracersFolder:ClearAllChildren()
    if State.ESP_Tracers then
        for _, P in ipairs(Players:GetPlayers()) do
            if P ~= LocalPlayer and P.Character and P.Character:FindFirstChild("HumanoidRootPart") then
                local ScreenPoint, OnScreen = Camera:WorldToScreenPoint(P.Character.HumanoidRootPart.Position)
                if OnScreen then
                    local Tracer = Drawing.new("Line")
                    Tracer.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                    Tracer.To = Vector2.new(ScreenPoint.X, ScreenPoint.Y)
                    Tracer.Color = State.ESP_Color
                    Tracer.Thickness = 1
                    Tracer.Transparency = 1
                    Tracer.Visible = true
                    task.spawn(function()
                        task.wait()
                        Tracer:Destroy()
                    end)
                end
            end
        end
    end
    
    if State.Spectate and State.SpectatedPlayer and State.SpectatedPlayer.Character and State.SpectatedPlayer.Character:FindFirstChildOfClass("Humanoid") then
        Camera.CameraSubject = State.SpectatedPlayer.Character:FindFirstChildOfClass("Humanoid")
    end
    
    if State.Desync and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local HRP = LocalPlayer.Character.HumanoidRootPart
        if State.DesyncType == "CFrame" then
            HRP.CFrame = HRP.CFrame * CFrame.new(math.random(-State.DesyncValue, State.DesyncValue)/10, 0, math.random(-State.DesyncValue, State.DesyncValue)/10)
        elseif State.DesyncType == "Velocity" then
            HRP.Velocity = Vector3.new(math.random(-1000, 1000), 0, math.random(-1000, 1000))
        elseif State.DesyncType == "Backwards" then
            HRP.CFrame = HRP.CFrame * CFrame.Angles(0, math.rad(180), 0)
        end
    end
    
    if FreecamActive and FreecamPart then
        local Frame = Camera.CFrame
        local Dir = Vector3.new()
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then Dir = Dir + Frame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then Dir = Dir - Frame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then Dir = Dir - Frame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then Dir = Dir + Frame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.E) then Dir = Dir + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.Q) then Dir = Dir - Vector3.new(0, 1, 0) end
        FreecamPart.CFrame = FreecamPart.CFrame + Dir * State.FreecamSpeed
        Camera.CFrame = FreecamPart.CFrame
    end
end)

task.spawn(function()
